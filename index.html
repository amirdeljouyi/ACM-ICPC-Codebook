DP

Unbounded Knapsack

Returns max value that you can fit inside of a given space. Assume not tested, from old codebook. Variables int N: Number of items int W: Max weight int[] v: Item values int[] w: Item weights

public static int knapsack(int N, int W, int[] v, int[] w) {
    int[] m = new int[W*N];

    for (int j = 0; j < W + 1; j++) {
        for (int i = 0; i < N; i++) {
            if (w[i] <= j) {
                m[j] = max(m[j], m[j-w[i]] + v[i]);
            }
        }
    }

    return m[W + (N-1)];
}
